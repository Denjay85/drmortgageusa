function LeadForm({ onOpenCalendly }) {
  const [state, setState] = useState({
    name: "",
    email: "",
    phone: "",
    yearBought: "",
    savingsGoal: "",
    currentRate: "",
    consent: false,
  });
  const [loading, setLoading] = useState(false);
  const [done, setDone] = useState(false);
  const [error, setError] = useState("");

  const disabled = useMemo(() => {
    return !state.name || !isEmail(state.email) || !isPhone(state.phone) || !state.yearBought || !state.savingsGoal || !state.consent;
  }, [state]);

  const onSubmit = async (e) => {
    e.preventDefault();
    setError("");
    setLoading(true);
    try {
      const payload = {
        source: "rate-watch",
        ...state,
        timestamp: new Date().toISOString(),
        utm: getUTM(),
      };
      const res = await fetch(POST_URL, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(payload),
      });
      if (!res.ok) throw new Error("Submit failed");
      setDone(true);
      // Optional Calendly open
      if (onOpenCalendly) onOpenCalendly();
      // Optional thank-you redirect after 90s
      setTimeout(() => {
        if (window.location.hash !== "#learn") window.location.href = "#learn";
      }, 90000);
    } catch (err) {
      setError("Could not submit. Try again or text 850-346-8514.");
    } finally {
      setLoading(false);
    }
  };

  if (done) return <Thanks />;

  return (
    <form onSubmit={onSubmit} className="mt-6 grid grid-cols-1 md:grid-cols-2 gap-4">
      <Field label="Full name" required>
        <input className="input" placeholder="Jane Doe" value={state.name} onChange={(e) => setState({ ...state, name: e.target.value })} />
      </Field>
      <Field label="Email" required>
        <input className="input" placeholder="you@email.com" value={state.email} onChange={(e) => setState({ ...state, email: e.target.value })} />
      </Field>
      <Field label="Phone" required>
        <input className="input" placeholder="(555) 555-5555" value={state.phone} onChange={(e) => setState({ ...state, phone: formatPhone(e.target.value) })} />
      </Field>
      <Field label="Year you bought" required>
        <input className="input" placeholder="2019" value={state.yearBought} onChange={(e) => setState({ ...state, yearBought: e.target.value.replace(/[^0-9]/g, "").slice(0,4) })} />
      </Field>
      <Field label="Monthly savings goal ($)" required>
        <input className="input" placeholder="200" value={state.savingsGoal} onChange={(e) => setState({ ...state, savingsGoal: e.target.value.replace(/[^0-9]/g, "") })} />
      </Field>
      <Field label="Current interest rate (optional)">
        <input className="input" placeholder="6.875" value={state.currentRate} onChange={(e) => setState({ ...state, currentRate: e.target.value.replace(/[^0-9.]/g, "") })} />
      </Field>

      <div className="md:col-span-2">
        <label className="flex items-start gap-3 text-sm">
          <input type="checkbox" checked={state.consent} onChange={(e) => setState({ ...state, consent: e.target.checked })} className="mt-1" />
          <span>
            You agree to be contacted by phone, text, and email at the info provided. Consent is not a condition of credit. Msg and data rates may apply.
          </span>
        </label>
      </div>

      {error && <p className="md:col-span-2 text-sm text-red-600">{error}</p>}

      <div className="md:col-span-2 flex items-center gap-3">
        <button type="submit" disabled={disabled || loading} className="btn-primary disabled:opacity-50">
          {loading ? "Submitting..." : "Start rate watch"}
        </button>
        <a href="#learn" className="btn-ghost">Learn more</a>
      </div>
    </form>
  );
}

// Helpers
function isEmail(v) {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(v.trim());
}
function isPhone(v) {
  const digits = v.replace(/\D/g, "");
  return digits.length >= 10;
}
function formatPhone(v) {
  const d = v.replace(/\D/g, "");
  const p1 = d.slice(0,3);
  const p2 = d.slice(3,6);
  const p3 = d.slice(6,10);
  if (d.length <= 3) return p1;
  if (d.length <= 6) return `(${p1}) ${p2}`;
  return `(${p1}) ${p2}-${p3}`;
}
function getUTM() {
  const params = new URLSearchParams(window.location.search);
  const keys = ["utm_source","utm_medium","utm_campaign","utm_content","utm_term"];
  const out = {};
  keys.forEach(k => { const v = params.get(k); if (v) out[k] = v; });
  return out;
}
