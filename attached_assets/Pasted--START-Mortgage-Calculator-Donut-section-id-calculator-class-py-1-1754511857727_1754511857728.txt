<script src="https://cdn.jsdelivr.net/npm/chart.js"></script>

<!-- Calculator + Chart logic -->
<script>
(() => {
  const $ = id => document.getElementById(id);
  const round = v => Math.round(v * 100) / 100;

  const colors = {
    pi:  '#005f7f',
    tax: '#8b004b',
    ins: '#3d8b37',
    pmi: '#d69800',
    hoa: '#c45100'
  };

  let chart;

  function buildChart(data) {
    const ctx = $('breakdownChart').getContext('2d');
    chart = new Chart(ctx, {
      type: 'doughnut',
      data: {
        labels: [
          'Principal & Interest',
          'Property Tax',
          "Homeowner's Insurance",
          'PMI',
          'HOA Fees'
        ],
        datasets: [{
          data,
          backgroundColor: [
            colors.pi,
            colors.tax,
            colors.ins,
            colors.pmi,
            colors.hoa
          ],
          hoverOffset: 4
        }]
      },
      options: {
        plugins: {
          legend: { position: 'right', labels: { boxWidth: 18 } },
          tooltip: { callbacks: { label: ctx => `$ ${round(ctx.parsed)}` } }
        },
        cutout: '60%'
      }
    });
  }

  function updateChart(data) {
    chart.data.datasets[0].data = data;
    chart.update();
  }

  function calc() {
    const price  = +$('price').value || 0;
    const down   = +$('down').value  || 0;
    const years  = +$('years').value || 30;
    const rate   = (+$('rate').value || 0) / 100 / 12;
    const tax    = (+$('tax').value  || 0) / 12;
    const ins    = (+$('ins').value  || 0) / 12;
    const hoa    = +$('hoa').value   || 0;
    const pmi    = 0; // replace with PMI logic if needed

    const n      = years * 12;
    const loan   = price - down;
    const pAndI  = rate === 0
                 ? loan / n
                 : loan * (rate / (1 - Math.pow(1 + rate, -n)));
    const total  = pAndI + tax + ins + pmi + hoa;

    $('result').innerHTML =
      `Principal &amp; Interest: $${round(pAndI).toLocaleString()}/mo<br>
       Taxes &amp; Insurance: $${round(tax + ins).toLocaleString()}/mo<br>
       HOA: $${round(hoa).toLocaleString()}/mo<br>
       <span class="text-[#001f3f]">Estimated Total Payment: $${round(total).toLocaleString()}/mo</span>`;

    const slices = [round(pAndI), round(tax), round(ins), round(pmi), round(hoa)];
    chart ? updateChart(slices) : buildChart(slices);
  }

  $('calcBtn').addEventListener('click', calc);
  calc(); // initial render
})();
</script>
